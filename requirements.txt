# LISA Windows Agent — Python Dependencies
# ==========================================
# Install with: pip install -r requirements.txt
#
# Required:
requests==2.31.0      # HTTP requests to the LISA server
pyyaml==6.0.1         # Reading YAML config and role files
psycopg2-binary==2.9.9 # Direct PostgreSQL connection (same as Linux agent)
pywin32==306          # Windows COM automation (Word, Excel, Outlook, registry)
                      # After installing pywin32, run:
                      # python -m win32com.client.makepy
                      # to generate type libraries for Office COM

# Optional but recommended:
pyautogui==0.9.54     # GUI automation fallback (typing, clicking)
python-docx==1.1.0    # Create/edit .docx files without opening Word
                      # (useful for testing without Office installed)
